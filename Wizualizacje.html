  1. Tworzenie najprostrzych wykresw
# Wywietlanie wykresu dla zbioru danych midwest. Na osi X pokazano odsetek osb
# z wyksztaceniem wyszym, a na osi Y wskanik ubstwa wrd dorosych.
ggplot(data = midwest) +
  geom_point(mapping = aes(x = percollege, y = percadultpoverty))

# Funkcja geom_ okrela rodzaj rysowanego wykresu
  1. Tworzenie wykresu kolumnowego
# Wykres supkowy ilustrujcy czn liczb ludnoci w kadym stanie.
# Kolumna state jest odwzorowana na o x, a kolumna poptotal na
# o y (o pionowa)
ggplot(data = midwest) +
  geom_col(mapping = aes(x = state, y = poptotal))

  1. Wykres heksagonalny
ggplot(data = midwest) +
  geom_hex(mapping = aes(x = percollege, y = percadultpoverty))

  1. Dodawanie dodatkowych obiektw do wykresu
# Wykres z punktami i wygadzon lini.
ggplot(data = midwest) +
  geom_point(mapping = aes(x = percollege, y = percadultpoverty)) +
  geom_smooth(mapping = aes(x = percollege, y = percadultpoverty))
## `geom_smooth()` using method = 'loess' and formula 'y ~ x'

# Wykres z punktami i wygadzon lini, dla ktrych zastosowano te same odwzorowania aspektw estetycznych.
ggplot(data = midwest, mapping = aes(x = percollege, 
                                     y = percadultpoverty)) +
  geom_point() + # Uywane s domylne odwzorowania dla osi x i y.
  geom_smooth() + # Uywane s domylne odwzorowania dla osi x i y.
  geom_point(mapping = aes(y = percchildbelowpovert)) # Uycie innego odwzorowania dla osi y.
## `geom_smooth()` using method = 'loess' and formula 'y ~ x'

#Dla wszystkich obiektw geometrycznych uywane s dane i aspekty estetyczne podane w funkcji ggplot(), chyba e oglne ustawienia zostay zastpione w poszczeglnych wywoaniach.
  1. Dodawanie dodatkowych aspektw do funkcji aes()
# Uzalenianie koloru kadego punktu od stanu, w jakim znajduje si dane hrabstwo.
ggplot(data = midwest) +
  geom_point(
    mapping = aes(x = percollege, y = percadultpoverty, color = state))

# Ustawienie jednego koloru ("red") dla wszystkich punktw (niezalenie od danych).
ggplot(data = midwest) +
  geom_point(
    mapping = aes(x = percollege, y = percadultpoverty),
    color = "red",
    alpha = .3)
